807  template<typename ...PointerTArgs>
808  MemBlock * allocate_(PointerTArgs&&... args)
809  {
810  // Return memory allocated here.
811  MemBlock * block = nullptr;
812 
813  // Check for previously freed blocks and reuse them.
814  if(free_idx_ > 0) {
815  --free_idx_;
816  block = free_blocks_[free_idx_];
817  sparta_assert(block->ref_count->p != nullptr);
818  block->ref_count->mem_block = (void*)block;
819  block->ref_count->count = 1;
820  // perform an in place new on the reused pointer
821  new (block->ref_count->p) PointerT(std::forward<PointerTArgs>(args)...);
822  }
823  else {
824  if(SPARTA_EXPECT_FALSE(allocated_ > water_mark_)) {
825  if(SPARTA_EXPECT_FALSE(!water_mark_warning_)) {
826  watermark_warning_callback_(*this);
827  water_mark_warning_ = true;
828  }
829 
830  // Only need to check for overallocation if we've passed the watermark
831  if(SPARTA_EXPECT_FALSE(allocated_ >= memory_blocks_.capacity())) {
832  SpartaException ex;
833  ex << "This allocator has run out of memory: \n\n\t"
834  << __PRETTY_FUNCTION__
835  << "\n\n"
836  << "\t\tNumber blocks preallocated: " << memory_blocks_.capacity()
837  << "\n\t\tWatermark : " << water_mark_;
838  throw ex;
839  }
840  }
841  block = memory_blocks_.allocate(this, std::forward<PointerTArgs>(args)...);
842  ++allocated_;
843  }
844  return block;
845  }

904  static void releaseRefCount_(RefCount *& ref_count)
905  {
906  if(0 == ref_count->count) {
907  BaseAllocator::MemBlockBase * memory_block =
908  static_cast<BaseAllocator::MemBlockBase *>(ref_count->mem_block);
909 
910  if(SPARTA_EXPECT_TRUE(nullptr != memory_block)) {
911  memory_block->alloc->releaseObject_(memory_block);
912  }
913  else {
914  delete ref_count->p;
915  ref_count->p = nullptr;
916  }
917  // Make it go negative to show there are no
918  // SpartaSharedPointer objects using this ref_count.
919  // We can't set it to nullptr because it might be
920  // allocated on the heap and we need to ensure all
921  // SpartaWeakPointer objects are done with it too.
922  --(ref_count->count);
923  }
924
