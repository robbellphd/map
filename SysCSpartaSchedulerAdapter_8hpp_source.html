43  inline Clock::Cycle calculateSpartaOffset(const sparta::Clock * sparta_clk,
44  sc_core::sc_time::value_type sysc_offset)
45  {
46  //
47  // This is a transaction coming from SysC that is on SysC's
48  // clock, not Sparta's. Need to find the same tick cycle on
49  // the Sparta clock and align the time for the transaction.
50  // Keep in mind that Sparta's scheduler starts on tick 1, not
51  // 0 like SysC.
52  //
53  // For example,
54  // - The Sparta's clock is at 7 ticks (6 from SysC POV, hence the - 1)
55  // - The SysC clock is at 10 ticks
56  // - The transaction's delay is 1 tick (to be fired at tick 11)
57  //
58  // sysc_clock - sparta_clock + delay = 4 cycles on sparta clock (11)
59  //
60 
61  // Send to memory with the given delay - NS -> clock cycles.
62  // The Clock is on the same freq as the memory block
63  auto current_sc_time = sc_core::sc_time_stamp().value();
64  const auto current_tick = sparta_clk->currentTick() - 1;
65  sparta_assert(sc_core::sc_time_stamp().value() >= current_tick);
66  const auto final_relative_tick = current_sc_time - current_tick + sysc_offset;
67  return final_relative_tick;
68  }

222  void runScheduler_()
223  {
224  // Start simulation -- align the schedulers
225  sparta_assert(sparta_scheduler_->nextEventTick() > 0);
226 
227  // Align the schedulers. Sparta starts at tick 1
228  wait(sc_core::sc_time(double(1), sparta_sc_time_));
229 
230  do {
231  // If the Sparta Scheduler has nothing to do, put it to sleep
232  if(sparta_scheduler_->nextEventTick() == Scheduler::INDEFINITE) {
233  sparta_scheduler_->registerForNotification<Scheduler::Tick,
234  SysCSpartaSchedulerAdapter,
235  &SysCSpartaSchedulerAdapter::wakeupAdapter_>(this, "item_scheduled");
236  wait(sc_wake_sparta_);
237  }
238 
239  //
240  // Wait for SysC to get to the next event time on the
241  // Sparta Scheduler, then advance Sparta
242  //
243  const sc_core::sc_time sysc_time = sc_core::sc_time_stamp();
244  if(sparta_scheduler_->nextEventTick() >= sysc_time.value()) {
245  // Wait until the sysc scheduled catches up with Sparta
246  wait(sc_core::sc_time(double(sparta_scheduler_->nextEventTick() - sysc_time.value()),
247  sparta_sc_time_));
248  }
249 
250  // Align to the posedge events in systemc
251  wait(sc_core::SC_ZERO_TIME);
252 
253  // If given, schedule the user's SystemC query event to
254  // allow the Sparta user to check to see if the SystemC side
255  // of simulation is complete. This query can be as simple
256  // as asking the SystemC kernel if there are any events
257  // (sc_core::sc_pending_activity()) or asking the SystemC
258  // simulator if it's done.
259  if(sysc_query_event_ && sparta_scheduler_->getCurrentTick() >= next_sysc_event_fire_tick_)
260  {
261  next_sysc_event_fire_tick_ = sparta_scheduler_->getCurrentTick() + sysc_query_event_interval_;
262  sysc_query_event_->scheduleRelativeTick(1, sparta_scheduler_);
263  }
264 
265  advanceSpartaScheduler_();
266 
267  } while(!sparta_scheduler_->isFinished() || !sysc_simulation_done_);
268 
269  // Stop simulation
270  sc_core::sc_stop();
271  sc_stop_called_ = true;
272  }

274  void advanceSpartaScheduler_()
275  {
276  const sc_core::sc_time sysc_time = sc_core::sc_time_stamp();
277 
278  // The SystemC scheduler will always be exactly at the same
279  // tick as Sparta when this function is called. Following
280  // this rule allows safe assumptions in scheduling
281  // synchronization.
282  sparta_assert(sysc_time.value() == sparta_scheduler_->nextEventTick());
283  sparta_assert(sparta_scheduler_->nextEventTick() >= sparta_scheduler_->getCurrentTick());
284 
285  const bool exacting_run = true;
286  const bool measure_scheduler_time = false; // no need to do this
287  // Run to the next event scheduled and then run that cycle
288  sparta_scheduler_->run(sparta_scheduler_->nextEventTick() - sparta_scheduler_->getCurrentTick() + 1,
289  exacting_run, measure_scheduler_time);
290 
291  // Sparta should now be finished with the cycle at `sysc_time`
292  // and be exactly one cycle ahead of it.
293  sparta_assert(sysc_time.value() + 1 == sparta_scheduler_->getCurrentTick());
294  }
